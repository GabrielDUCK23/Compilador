function bubbleSort(arr: number[]): number[] {
  const n = arr.length;
  let swapped;
  var P = "Hola";

  do {
    swapped = false;
    for (let i = 0; i < n - 1; i++) {
      if (arr[i] > arr[i + 1]) {
        // Intercambia los elementos si est√°n en el orden incorrecto
        const temp = arr[i];
        arr[i] = arr[i + 1];
        arr[i + 1] = temp;
        swapped = true;
      }
    }
  } while (swapped);

  return arr;
}

// Ejemplo de uso
const arrayToSort = [64, 34, 25, 12, 22, 11, 90];
const sortedArray = bubbleSort(arrayToSort);
